1   package logic.database;

3   import com.mongodb.MongoClient;
4   import com.mongodb.MongoClientOptions;
5   import com.mongodb.MongoCredential;
6   import com.mongodb.ServerAddress;
7   import com.mongodb.client.MongoDatabase;
8   import config.Configuration;
9   
10  import java.util.Arrays;
11  import java.util.Properties;
12  
13  /**
14   * Class to connect the application logic to a MongoDB server.
15   * Using Configuration to get credentials for database connection.
16   */
17  public class Database {
18  
19      private Properties properties;
20  
21      /**
22       * Initializes the class variable with passed in Properties object.
23       *
24       * @param properties Properties to be loaded in the class variable.
25       */
26      public Database(Properties properties) {
27          this.properties = properties;
28      }
29  
30      /**
31       * Initializes the database connection by using the configurations
32       * passed on to the constructor earlier.
33       *
34       * @return MongoClient database connection.
35       */
36      public MongoClient connection() {
37          MongoCredential credentials = MongoCredential.createCredential(
38                  properties.getProperty("user-name"),
39                  properties.getProperty("db-name"),
40                  properties.getProperty("password").toCharArray());
41          MongoClientOptions options = MongoClientOptions.builder().sslEnabled(true).build();
42          return new MongoClient(new ServerAddress(properties.getProperty("db-hostname"), Integer.parseInt(properties.getProperty("db-port"))), Arrays.asList(credentials), options);
43      }
44  
45      /**
46       * Gets the database by name set in the properties by using a database connection
47       * passed from connection() method.
48       *
49       * @return MongoDatabase object to be used in database transactions.
50       */
51      public MongoDatabase getDatabase() {
52          return connection().getDatabase(this.properties.getProperty("db-name"));
53      }
54  
55  }
