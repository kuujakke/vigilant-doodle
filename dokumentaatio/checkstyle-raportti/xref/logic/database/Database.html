3   import com.mongodb.*;
4   
5   import config.Configuration;
6   import config.DefaultSettings;
7   import org.mongodb.morphia.Datastore;
8   import org.mongodb.morphia.Morphia;
9   
10  import java.util.Arrays;
11  
12  import java.util.Properties;
13  
14  /**
15   * Class to connect the application logic to a MongoDB server.
16   * Using Configuration to get credentials for database connection.
17   */
18  public class Database {
19  
20      private Properties properties;
21      private Morphia morphia = new Morphia();
22      private Datastore datastore;
23  
24      /**
25       * Initializes the class variable with passed in Properties object.
26       *
27       * @param properties Properties to be loaded in the class variable.
28       *
29       * @throws Exception if invalid properties.
30       */
31      public Database(Properties properties) throws Exception {
32          if (properties != null) {
33              this.properties = properties;
34          } else {
35              this.properties = new Configuration().getProperties();
36          }
37          this.morphia.mapPackage("fi.jk.vigilant-doodle");
38      }
39  
40      /**
41       * Initializes the database connection by using the configurations
42       * passed on to the constructor earlier.
43       *
44       * @return MongoClient database connection.
45       */
46      public MongoClient connection() {
47          if (!this.properties.isEmpty() && this.properties != null) {
48              MongoCredential credentials = MongoCredential.createCredential(
49                      this.properties.getProperty("db-user"),
50                      this.properties.getProperty("db-name"),
51                      this.properties.getProperty("db-password").toCharArray());
52              MongoClientOptions options = MongoClientOptions.builder().sslEnabled(false).build();
53              try {
54                  return new MongoClient(new ServerAddress(properties.getProperty("db-hostname"), Integer.parseInt(properties.getProperty("db-port"))), Arrays.asList(credentials), options);
55              } catch (Exception e) {
56                  e.printStackTrace();
57              }
58          }
59          return null;
60      }
61  
62      /**
63       * Gets the database by name set in the properties by using a database connection
64       * passed from connection() method.
65       *
66       * @return Datastore object to be used in database transactions.
67       */
68      public Datastore getDatabase() {
69          if (this.datastore == null) {
70              this.datastore = this.morphia.createDatastore(connection(), this.properties.getProperty("db-name"));
71              this.datastore.ensureIndexes();
72          }
73          return datastore;
74      }
75  
76  }
