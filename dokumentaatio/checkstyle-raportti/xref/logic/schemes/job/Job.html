1   package logic.schemes.job;

3   import logic.schemes.Scheme;
4   import logic.roles.Role;
5   
6   /**
7    * Base object to keep track of a job being done in a task.
8    * Holds one worker that is responsible for completing the job.
9    */
10  public class Job extends Scheme {
11  
12      private Role worker;
13  
14      /**
15       * Passes the name to the superclass.
16       *
17       * @param name String containing the name of the job.
18       */
19      public Job(String name) {
20          super(name);
21      }
22  
23      @Override
24      public boolean hasScheme(Scheme scheme) {
25          return this.equals(scheme);
26      }
27  
28      /**
29       * Returns the worker responsible on the completion of the job.
30       *
31       * @return Role that is assigned to the job.
32       */
33      public Role getWorker() {
34          return this.worker;
35      }
36  
37      /**
38       * Sets the given worker to be responsible on the completion of the job.
39       *
40       * @param worker Role that is to be assigned to the job.
41       */
42      public void setWorker(Role worker) {
43          this.worker = worker;
44          this.worker.addResponsibility(this);
45      }
46  }
