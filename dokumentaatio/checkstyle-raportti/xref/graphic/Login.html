19      public Login() {

23      /**
24       * Used to popup a JOptionPane to ask user for login information.
25       * Sets up a jPanel layout and adds labels, text boxes and a login button.
26       * It asks for Username, password, database name and database address.
27       * Next it casts a JOptionPane to get user input and returns the Propeties
28       * object containing the login information.
29       *
30       * @return Properties object containing users login information.
31       */
32      public Properties getLoginInformation() {
33          HashMap<String, JComponent> components = makelayout();
34          int result = JOptionPane.showConfirmDialog(null, this, "Please login to proceed", JOptionPane.OK_CANCEL_OPTION);
35          if (result == JOptionPane.OK_OPTION) {
36              this.loginInformation.setProperty("canceled", "0");
37              this.loginInformation.setProperty("user-name", ((JTextField) components.get("username")).getText());
38              char[] password = ((JPasswordField) components.get("password")).getPassword();
39              this.loginInformation.setProperty("password", String.valueOf(password));
40              this.loginInformation.setProperty("db-name", ((JTextField) components.get("db-name")).getText());
41              this.loginInformation.setProperty("db-hostname", ((JTextField) components.get("db-hostname")).getText());
42              this.loginInformation.setProperty("db-port", ((JTextField) components.get("db-port")).getText());
43              return this.loginInformation;
44          } else if (result == JOptionPane.CANCEL_OPTION) {
45              Properties props = new Properties();
46              props.setProperty("canceled", "1");
47              return props;
48          }
49          return null;
50      }

144     /**
145      * Sets basic constraints for given position in the GridBagLayout.
146      * @param c GridBagConstraints
147      * @param gridx Grid X coordinate
148      * @param gridy Grid Y coordinate
149      */
150     private void setConstraints(GridBagConstraints c, int gridx, int gridy) {
151         c.fill = GridBagConstraints.CENTER;
152         c.weightx = 0.5;
153         c.gridx = gridx;
154         c.gridy = gridy;
155     }
156 
157     /**
158      * Validates the information that user has given.
159      * Checks that properties doesn't have "canceled" set to true before continuing
160      * the validation by checking that each property has a minimum length of 3 characters.
161      *
162      * @param props The Properties object which has the users input stored.
163      * @return a truth value whether the input is valid or not.
164      */
165     public boolean validateProps(Properties props) {
166         int validStrings = 0;
167         for (String prop : props.stringPropertyNames()) {
168             if (props.getProperty("canceled").equals("1")) {
169                 return false;
170             } else if (props.getProperty(prop).length() > 2) {
171                 validStrings++;
172             }
173         }
174         if (validStrings == 5) {
175             return true;
176         }
177         return false;
178     }

180     /**
181      * Creates horisontal strut to the grid at given gridy value.
182      *
183      * @param width Set width of the strut.
184      * @param gridy Set the grid position of the strut.
185      * @param c     GridBagConstraints
186      */
187     private void horisontalStrut(int width, int gridy, GridBagConstraints c) {
188         c.fill = GridBagConstraints.CENTER;
189         c.weightx = 0.5;
190         c.gridx = 0;
191         c.gridy = gridy;
192         add(Box.createVerticalStrut(width), c);
193     }
